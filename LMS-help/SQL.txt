SELECT
A.tx_loan_tracking_id
, ca.tx_bp_no
, A.tx_application_no
, A.dec_applied_loan_amount
, a.dec_recommended_for_approval
, A.dtt_mod
, A.id_legal_entity_key
, A.id_creator_key
, A.dtt_create
, U.tx_login_name
, st.tx_state_name
, cl.tx_value1
, cc.tx_value1
, ca.dtt_date_of_birth
, ca.tx_age
, ca.dtt_joining_date
, ca.tx_service_length
FROM T_LOAN A
join T_CONFIGURATION cl on cl.id_configuration_key = a.id_loan_type_key
join T_CONFIGURATION cc on cc.id_configuration_key = a.id_customer_type_key
JOIN T_USER U ON A.id_user_mod_key=U.id_user_key
Join T_CUSTOMER ca on A.id_customer_key=ca.id_customer_key
JOIN T_FSM_STATE ST ON ST.id_fsm_state_key=A.id_state_key WHERE A.tx_data_source='FILE'
AND A.dtt_create='2020-02-24'
